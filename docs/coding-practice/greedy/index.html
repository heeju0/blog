#18238 ZOAC 2#

메모리시간코드 길이
28776 KB68 ms133 B
입력#
ZOAC
출력#
26
내 풀이#
word = 'A' + input()
res = 0
for i, j in zip(word[:-1], word[1:]):
x = abs(ord(j)-ord(i))
res += min(x, abs(x-26))
print(res)
다른 풀이#
before = 0
res = 0
for i in map(ord, input()):
i -= 65
res = res + min((before - i)%26, (i - before)%26)
before = i
print(res)

#2839 설탕 배달#

메모리시간코드 길이
28776 KB76 ms197 B
입력#
18
출력#
4
내 풀이#
n = int(input())
res = -1
for j in range(n//3 + 1):
for i in range(n//5 + 1):
if 5 * i + 3 * j == n:
res = i + j
break
if res != -1:
break
print(res)
다른 풀이#
n = int(input())
a = n//5
b = n%5
res = -1
while a >= 0:
if b%3 == 0:
res = a + b//3
break
a -= 1
b += 5
print(res)
