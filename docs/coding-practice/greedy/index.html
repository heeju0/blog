T = int(input())
def flip(s, k):
idx = s.find('-')
s = list(s)
if idx != -1 and idx+k <= len(s):
s[idx:idx+k] = map(lambda x: '+' if x == '-' else '-',s[idx:idx+k])
return ''.join(s)
for t in range(T):
s, k = input().split()
k = int(k)
res = [s]
while True:
s = flip(s, k)
if s in res:
break
else:
res.append(s)
if '+'*len(s) in res:
break
print('Case #{}: {}'.format(t+1, len(res)-1 if '+'*len(s) in res else 'IMPOSSIBLE'))
다른 풀이#

그 다음 남은 공이 있으면 1, 없으면 0을 더해주었다.

n, k = map(int, input().split())
x = sum(list(range(1, k+1)))

res = k - 1 + b
if x > n:
res = -1
print(res)

$\sum\limits_{i=0}^{k}{i}$ - 이 식을 sum(list(range(1, k+1))) 대신 k*(k+1)/2로 구현

(왜 이렇게 복잡하게 생각했을까?)
