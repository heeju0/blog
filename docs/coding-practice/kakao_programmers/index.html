내 풀이 1#

시간 초과 문제로 테스트 10을 통과할 수가 없다.

def solution(enroll, referral, seller, amount):
answer = {i:0 for i in enroll}
dic = {}
for i, j in zip(referral, enroll):
dic[i] = dic.get(i, []) + [j]
for s, a in zip(seller, amount):
dfs(s, a*100, answer, dic)
return list(answer.values())
def dfs(seller, amount, answer, dic):
for i in dic:
if seller in dic[i]:
next = (amount * 0.1)//1
answer[seller] += amount - next
return dfs(i, next, answer, dic)

def solution(s):
N = len(s)
answer = N
# N/2회 반복
for i in range(1, N//2 + 1):
# s를 i 길이씩 분할
lst = [s[j:j+i] for j in range(0, N, i)]
# res: 결과 저장, temp: 현재 문자열, cn: 문자열 카운트
res, temp, cn = '', lst[0], 1
for k in lst[1:]:
# 문자열이 재등장하면 카운트
if temp == k:
cn += 1
# 다른 문자열이 등장하면 res에 저장하고 temp와 cn 초기화
else:
res += (str(cn) + temp) if cn != 1 else temp
temp, cn = k, 1
# 남은 문자열을 res에 추가
if res[-i:] != temp:
res += (str(cn) + temp) if cn != 1 else temp
answer = min(answer, len(res))
return answer

처음에 eval() 메서드를 생각 못해서 시간이 오래걸렸다. 이런 사랑스러운 함수가 다있다니..😆

operator = [
('*', '+', '-'),
('*', '-', '+'),
('+', '*', '-'),
('+', '-', '*'),
('-', '*', '+'),
('-', '+', '*')
]
def solution(expression):
answer = 0
for o in operator:
exp_lst = []
for i in [i.split(o[1]) for i in expression.split(o[2])]:
exp_lst.append([j.split(o[0]) for j in i])
for i, x in enumerate(exp_lst):
for j in range(len(x)):
if len(exp_lst[i][j]) > 1:
exp_lst[i][j] = str(eval(o[0].join(exp_lst[i][j])))
else:
exp_lst[i][j] = exp_lst[i][j][0]
if len(x) > 1:
exp_lst[i] = str(eval(o[1].join(exp_lst[i])))
else:
exp_lst[i] = exp_lst[i][0]
res = abs(eval(o[2].join(exp_lst)))
answer = max(answer, abs(res))
return answer

def solution(expression):
def calc(p, n, expression):
if n == 2:
return str(eval(expression))
else:
res = eval(p[n].join([calc(p, n+1, e) for e in expression.split(p[n])]))
return str(res)
priorities = [
('*', '+', '-'),
('*', '-', '+'),
('-', '+', '*'),
('-', '*', '+'),
('+', '*', '-'),
('+', '-', '*'),
]
answer = 0
for p in priorities:
res = int(calc(p, 0, expression))
answer = max(answer, abs(res))
return answer

answer = []
for i in room_number:

내 풀이 2 - 성공#

find 함수의 재귀를 구현하는데 조금 머리 아팠다. 재귀를 위해 return 값을 추가로 넣었다.

import sys
sys.setrecursionlimit(10**6)
def solution(k, room_number):
dic = {}
for i in room_number:
find(i, dic)
return list(dic)
def find(n, dic):
if n not in dic:
dic[n] = n + 1
return n + 1
else:
dic[n] = find(dic[n], dic)
return dic[n]
