def func(p, b):
a = len(p)
if a < b:
return (p * (b//a + 1))[:b]
else:
return p[:b]
def compare(ans, *p):
res = []
for i in p:
cn = 0
for a, b in zip(i, ans):
if a == b:
cn += 1
res.append(cn)
return res

def solution(answers):
p = [[1, 2, 3, 4, 5],
[2, 1, 2, 3, 2, 4, 2, 5],
[3, 3, 1, 1, 2, 2, 4, 4, 5, 5]]
score = [0, 0, 0]
res = []
for i, x in enumerate(answers):
for n, a in enumerate(p):
if x == a[i%len(a)]:
score[n] += 1
for i, s in enumerate(score):
if s == max(score):
res.append(i + 1)
return res
