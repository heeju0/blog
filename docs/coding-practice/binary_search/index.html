x = list(map(int, input().split()))
res = max(n-x[-1], x[0])
for i, j in zip(x, x[1:]):
res = max(res, int((j-i)/2) if ((j-i)/2)%1 == 0 else round((j-i)/2 + 0.5))
print(res)
