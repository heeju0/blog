binary(x, lst, start, m-1)

from math import ceil
import sys
input = sys.stdin.readline
N, M = map(int, input().split())
lst = sorted(int(input()) for _ in range(M))
def binary(x, tot, lst, start, end):
if start > end:
return x
m = (start + end)//2
cn = 0
for i in lst:
cn += ceil(i/m)
if cn > tot:
return binary(x, tot, lst, m+1, end)
else:
return binary(m, tot, lst, start, m-1)

import sys
input = sys.stdin.readline
S, C = map(int, input().split())
L = [int(input()) for _ in range(S)]
def binary(x, tot, res, lst, start, end):
if start > end:
return sum(lst) - res*tot
m = (start + end)//2
cn = 0
for i in lst:
cn += i//m
if cn == tot + 1 and m * (tot + 1) == sum(lst):
return sum(lst) - m*tot
if cn >= tot:
return binary(x, tot, m, lst, m+1, end)
else:
return binary(x, tot, m, lst, start, m-1)
print(binary(max(L), C, 0, L, 1, max(L)))
