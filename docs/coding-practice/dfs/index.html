def bfs(x, y):
q = [(x, y)]
visited[x][y] = 1
while q:
x, y = q.pop(0)
for i in range(8):
dx = x + X[i]
dy = y + Y[i]
if 0 <= dx < M and 0 <= dy < N and not visited[dx][dy] and grid[dx][dy]:
visited[dx][dy] = 1
q.append((dx, dy))
cn = 0
for i in range(M):
for j in range(N):
if grid[i][j] and not visited[i][j]:
cn += 1
bfs(i, j)
print(cn)
